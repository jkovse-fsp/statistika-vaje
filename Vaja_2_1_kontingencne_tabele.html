# Check for normality of continuous variables
for var in ['starost', 'tedenski_km', 'povp_hitrost']:
    variable = df[var]
    stat, p = shapiro(variable)
    print(f'Shapiro-Wilk test statistic for {var}: {stat:.4f}')
    print(f'                              p-value: {p:.4f}')
    if p < 0.05:
        print(f'Variable **{var}** is not normally distributed. Using MEDIAN to halve it.')
        print(f'') 
        median = variable.median()
        df[f'{var}<{median:.1f}'] = pd.Categorical(np.where(df[var] < median, f'< {median:.1f}', f'> {median:.1f}'))
    else:
        print(f'Variable **{var}** is normally distributed. Using MEAN to halve it.')
        print('')
        mean = variable.mean()
        df[f'{var}<{mean:.1f}'] = pd.Categorical(np.where(df[var] < mean, f'< {mean:.1f}', f'> {mean:.1f}'))

results = []
df['poskodba_da_ne'] = df['poskodba_da_ne'].astype('category')
for category in ['spol', 'BMI<26', 'stopalo', 'izkusnje10', 'ogrevanje', 'vaje', 'starost<44.6', 'tedenski_km<21.1', 'povp_hitrost<11.2']:
    table = pd.crosstab(df[category], df['poskodba_da_ne'])
    out, observed, expected = pg.chi2_independence(data=df, x=category, y='poskodba_da_ne')
    
    res = expected.iloc[0]
    
    if res['pval'] < 0.05:
        print(f'=====================================================\nChi-square test for {category} vs poskodba_da_ne')
        print('Contingency table:\n', table)
        print(f'Chi2: {res['chi2']:.2f},     p-value: {p:.4f},      Degrees of Freedom: {res['dof']}\n')
        
        print(f'====> Significant association with {category}!')
        df[category] = df[category].astype('category')
        # calculate odds_ratio
        odds = odds_ratio(table)
        stat = odds.statistic
        low, high  = odds.confidence_interval(confidence_level=.95)

        print(f'\nObeti za poškodbo glede na {category}: {stat:.2f} (CI={low:.2f}, {high:.2f})\n')
        if stat < 1:
            print(f'Možnost poškodbe je manjša, če smo izpostavljeni faktorju {category}')
        else:
            print(f'Možnost poškodbe je večja, če smo izpostavljeni faktorju {category}')
    
        sns.countplot(x=category, hue='poskodba_da_ne', data=df)
        plt.title(f'Poskodba glede na {category}')
        plt.show()
    
        results.append({
            'Spremenljivka': category,
            'Chi2': res['chi2'],
            'p-value': f"{res['pval']:.3f}",
            'dof': res['dof'],
            'Cramer\'s V': res['cramer'],
            'Odds ratio (CI95%)': f'{stat:.3f} ({low:.3f}, {high:.3f})'
        })

results_df = pd.DataFrame(results)
results_df.style.set_caption(f'Table 1: Rezultati testov kontingenčne tabele')\
          .set_table_styles([
              {'selector': 'caption', 'props': [('font-size', '16px'),
